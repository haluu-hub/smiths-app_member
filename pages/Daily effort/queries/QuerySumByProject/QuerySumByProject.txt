SELECT p.project_name x, sum(t.actual_hours) y FROM public."timesheet" t 
LEFT JOIN member m on m.member_id = t.member_id 
LEFT JOIN project p on p.project_id = t.project_id 
WHERE m.email = '{{appsmith.user.email}}'
AND t.working_date BETWEEN '{{FromDatePicker.selectedDate}}' AND '{{ToDatePicker.selectedDate}}'
GROUP BY p.project_name
LIMIT {{data_table.pageSize}}
OFFSET {{(data_table.pageNo - 1) * data_table.pageSize}};
